Codex 技术规格书 · 完整架构

一、项目定义做什么一个浏览器单页应用，引导用户从 Idea 走到可以直接喂给 novel-workflow-refactor 的章节细化 TXT。中间产出的所有资产（角色卡、世界观、总纲、卷纲）可管理、可编辑、可导出。
不做什么不做写骨、润色、质检——那是下游 novel-workflow-refactor 的事。不做 RAG/记忆系统——那是 VCP Agent 的事。本工具只负责"规划"和"资产管理"。
技术约束· 纯浏览器，无 Node 后端，无构建工具，无 npm· 所有 JS 通过 <script> 标签顺序加载，全局作用域，和下游工具技术栈一致· 共享下游的 API 网关 localhost:6005，OpenAI 兼容格式· localStorage 持久化，key 前缀 nvpl_（区分下游的 novel.workflow.）· 视觉风格复用下游的 CSS 变量体系（--bg, --panel, --accent 等）
对接方式最终输出一份 UTF-8 TXT → 用户复制/下载 → 粘贴到 novel-workflow-refactor 的"多章节细化文本"框 → 点"拆分章节任务" → 跑三阶段流水线。
省钱设计每步一次 API 调用。不做多轮对话。AI 直接出完整结果，用户在本地编辑器中改，改完确认进下一步。

/**
 * 调用模型。支持 SSE 流式和非流式。
 * @param {string} systemPrompt - system message 内容
 * @param {string} userMessage  - user message 内容
 * @param {function} onChunk    - 流式回调 (deltaText) => void
 * @returns {Promise}   - 完整响应文本
 */
async function callModel(systemPrompt, userMessage, onChunk = null) {
  const cfg = NP.config;
  const abort = new AbortController();
  NP.activeAbort = abort;  const payload = {    model:       cfg.model,    temperature: cfg.temperature,    max_tokens:  cfg.maxTokens,    stream:      cfg.enableStream,    messages: [      { role: "system", content: systemPrompt },      { role: "user",   content: userMessage },    ],  };
  const resp = await fetch(cfg.apiUrl, {    method:  "POST",    headers: {      "Content-Type":  "application/json",      "Authorization": Bearer ${cfg.apiKey},    },    body:   JSON.stringify(payload),    signal: abort.signal,  });
  if (!resp.ok) {    const body = await resp.text().catch(() => "");    throw new Error(HTTP ${resp.status}: ${body.slice(0, 200)});  }
  if (!cfg.enableStream) {    const json = await resp.json();    const text = json.choices?.[0]?.message?.content || "";    if (onChunk) onChunk(text);    return text;  }
  // SSE 流式解析——复用下游 parseSseChunk 的模式  const reader = resp.body.getReader();  const decoder = new TextDecoder();  let full = "";  let buffer = "";
  while (true) {    const { done, value } = await reader.read();    if (done) break;    buffer += decoder.decode(value, { stream: true });
const lines = buffer.split("\n");
buffer = lines.pop() || "";

for (const line of lines) {
  const trimmed = line.trim();
  if (!trimmed || trimmed === "data: [DONE]") continue;
  if (!trimmed.startsWith("data: ")) continue;
  try {
    const json = JSON.parse(trimmed.slice(6));
    const delta = json.choices?.[0]?.delta?.content || "";
    if (delta) {
      full += delta;
      if (onChunk) onChunk(delta);
    }
  } catch (e) {
    // 忽略解析错误的行
  }
}

  }
  NP.activeAbort = null;  return full;}

/**

构建 user message。

根据步骤 ID 和上下文数据，返回完整的 user message 字符串。

每步的构建逻辑不同——这个函数是 30-wizard.js 中 generateStep 的核心依赖。 */function buildUserMessage(stepId, context = {}) {  switch (stepId) { case 1: // 选题定盘   return 我的故事想法：\n\n${NP.idea}\n\n请直接输出完整选题定盘书。;
 case 2: // 角色设计   return [ 选题定盘书：\n${NP.topicSheet}, \n已有角色：\n${NP.characters.map(c => · c.name（{c.name}（c.name（{roleLabel(c.role)}）).join("\n") || "暂无"}, \n请为以下角色生成完整资产卡：, 角色定位：${roleLabel(context.role)}, 角色名：${context.name || "请你起名"}, 补充描述：${context.desc || "无"},   ].join("\n");

function roleLabel(role) {  return { protagonist: "主角", heroine: "女主", antagonist: "反派", supporting: "配角" }[role] || role;}
function extractCharacterSummary(content) {  // 从角色卡中提取：姓名、身份锚点、能力起点、关系网  // 省略全文，只发摘要，省 token  const lines = content.split("\n");  const keep = [];  let capturing = false;  const captureKeys = ["姓名", "身份锚点", "角色定位", "核心命题", "能力", "起点", "关系网"];  for (const line of lines) {    if (captureKeys.some(k => line.includes(k))) { capturing = true; }    if (capturing) {      keep.push(line);      if (keep.length > 12) break;    }  }  return keep.join("\n") || content.slice(0, 500);}
function filterCharactersForVolume(volumeNumber) {  // 尝试从母总纲的"角色出场时间表"中判断本卷出场角色  // 如果判断不了就返回全部角色  // 简单实现：搜索 "卷{volumeNumber}" 出现在哪些角色行  const table = NP.masterOutline.match(/角色出场时间表[\s\S]?(?=\n\s伏笔|===|$)/)?.[0] || "";  if (!table) return NP.characters.map(c => c.content).join("\n\n---\n\n");
  const volPattern = new RegExp(卷${volumeNumber}|卷[${cn(volumeNumber)}], "g");  const relevant = NP.characters.filter(c => {    const line = table.split("\n").find(l => l.includes(c.name));    return !line || volPattern.test(line);  });  return (relevant.length ? relevant : NP.characters).map(c => c.content).join("\n\n---\n\n");}
function cn(n) {  return "零一二三四五六七八九十"[n] || n;}
